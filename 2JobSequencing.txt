#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// Function to perform Job Sequencing
vector<int> jobSequencing(vector<int> &deadline, vector<int> &profit) {
    int n = deadline.size();
    int cnt = 0;         // Count of jobs done
    int totalProfit = 0; // Total profit earned

    // Step 1: Pair profit and deadline together
    vector<pair<int, int>> jobs;
    for (int i = 0; i < n; i++) {
        jobs.push_back({profit[i], deadline[i]});
    }

    // Step 2: Sort jobs in decreasing order of profit
    sort(jobs.begin(), jobs.end(), greater<pair<int, int>>());

    // Step 3: Find the maximum deadline
    int maxDeadline = 0;
    for (auto &job : jobs)
        maxDeadline = max(maxDeadline, job.second);

    // Step 4: Create empty slots (0 = free)
    vector<int> slot(maxDeadline, 0);

    // Step 5: Assign jobs greedily
    for (int i = 0; i < n; i++) {
        int profit = jobs[i].first;
        int deadline = jobs[i].second;

        // Find the latest available slot before the deadline
        for (int j = min(maxDeadline, deadline) - 1; j >= 0; j--) {
            if (slot[j] == 0) {
                slot[j] = 1; // Mark slot as occupied
                cnt++;
                totalProfit += profit;
                break;
            }
        }
    }

    // Return {number of jobs done, total profit}
    return {cnt, totalProfit};
}

int main() {
    // Example input
    vector<int> deadline = {2, 1, 2, 1, 1};
    vector<int> profit = {100, 19, 27, 25, 15};

    vector<int> ans = jobSequencing(deadline, profit);

    cout << "Number of Jobs done: " << ans[0] << endl;
    cout << "Total Profit: " << ans[1] << endl;

    return 0;
}
